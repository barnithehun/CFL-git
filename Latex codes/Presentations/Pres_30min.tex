\begin{frame}[label=motivation1]
    \frametitle{Motivation}

Collateral constraints are major impediments to firm growth \footnotesize  - Beck et al. (2006)  \small
\begin{itemize} \small
    \item Yield credit misallocation and output losses \footnotesize - Catherine et al. (2021)
\end{itemize} \normalsize  \vspace{5mm} \\

Cash flow-based (CF-based) lending could alleviate collateral constraints 
\begin{itemize} \small
    \item \textbf{Backed by future cash flows} rather than physical assets    \hfill \hyperlink{LandM}{\beamergotobutton{Details}}

\begin{itemize} \small
    \item Redistributes household savings to firms in one-period debt contracts
    \item Offers interest rate to firms s.t. a zero profit condition
\end{itemize} \normalsize \vspace{2mm}

\pause

- \textbf{Bankruptcy Courts}: decide liquidation or reorganization

\begin{frame}[label=LiqReorg] \frametitle{Liquidation and Reorganization  \hfill \hyperlink{LiqReorg_backup}{\beamergotobutton{Details}}} 


\textbf{Liquidation}
\begin{itemize} \small
    \item The firm exits and its capital is resold at a discount (asset-specificity)
    \item Retained value: the resale value of assets
\end{itemize} \normalsize 


\textbf{Reorganization}
\begin{itemize} \small
    \item  The debt is renegotiated at a variable and a \textbf{fixed cost} \hfill \hyperlink{FCreorg}{\beamergotobutton{Reorganization Costs}}
    \item  The lender receives a transfer from the household, the firm continues
    \item  Retained value: going-concern value of the firm net of reorg. costs 
\end{itemize} \normalsize \vspace{3mm}

Default resolution by bankruptcy courts: maximizes value retanined in bankruptcy
$$ \underbrace{\text{Firm value} - \text{Reorganization costs}}_{\textcolor{red}{Reorganization}} \qquad \lessgtr \qquad \underbrace{\text{Resale value of assets}}_{\textcolor{red}{Liquidation}}  $$

\end{frame}

\begin{frame}[label=mechanism] \frametitle{Summary of the Model Mechanism}

CF-based debt is exposed to liquidation risk, asset-based debt is not 

Lender considers,
\begin{itemize} \small
    \item Ex-ante probability of default \textit{and} liquidation in default
    \item Expected pay-offs under liquidation or reorganization, given debt contract
    \end{itemize} \normalsize \vspace{3mm} 


Economies of scale in default resolution
\begin{itemize} \small
    \item Fixed reorganization cost: small firms often liquidate, large firms mostly reorganize 
\end{itemize} \normalsize \vspace{3mm} 

Misallocation of credit:
\begin{itemize} \small
    \item Ex-ante liquidation risk limits small firms' access to CF-based debt
    \item Small but productive firms (young) firms operate below efficient scale 
\end{itemize} \normalsize


\end{frame}

\begin{frame}[label = LandM] \frametitle{Cash Flow-Based and Asset-Based Lending \hfill \hyperlink{motivation1}{\beamergotobutton{Back}}}

Asset-based lending:
\begin{itemize} \small
\item Secured against borrowers' physical  assets
\item Creditors in-default payoffs:\textbf{ value of physical collateral}
\end{itemize} \normalsize 
Cash flow-based lending:
\begin{itemize} \small
\item No specific assets pledged as collateral
\item Secured against the entire corporate entity, or unsecured
\item Creditors in-default payoffs:\textbf{ expected future cash flows}
\end{itemize} \normalsize
Lian and Ma (2021): 80\% of US corporate debt is backed by future cash-flows \vspace{3mm}

Different from security: 
\begin{itemize} \small
    \item Secured vs. Unsecured: priority in bankruptcy
    \item Asset vs CF-based: economic determinants of lenders' in-default payoffs
\end{itemize} \normalsize

\end{frame}

\begin{frame}[label=Household] \frametitle{Representative Household \hfill \hyperlink{Model Summary}{\beamergotobutton{Back}}}

Unit measure of identical households, choose
\begin{itemize} \small
    \item Stream of consumption $C$ 
    \item One-period, bonds $B$ which yield the risk free interest rate $q_0$
\end{itemize} \normalsize

$$ V_h(B, \mu_0) = \max_{C,B'} U(C) + \beta V_h(B',\mu_0') $$
subject to: 
$$ C + B \leq w N^s + q_0 B' + T $$
where $N^s$ is the labour supply, $\beta$ is the discount factor \vspace{3mm} \\
$T$ consolidates expenses and revenues from the firm sector

\end{frame}

%------------------------------------------------
\begin{frame}[label=HH_T] \frametitle{Representative Household - The Transfer \hfill \hyperlink{Lpayoff}{\beamergotobutton{Back}}}

$T$ consolidates expenses and revenues from the firm sector \vspace{-5mm} 

\begin{equation*} \label{eq:T}
    \begin{split}
        & T = \int (1 - \chi_l - \chi_{ex}) \left( x - k'(k,b,\varepsilon) + qb'(k,b,\varepsilon) \right) d \mu_0(k,b,\varepsilon) \\
        & \qquad + \int \chi_{ex} \ x \ d \mu_0(k,b,\varepsilon) \\
        & \qquad + \int \chi_d \chi_{l} [ \phi_a (1-\delta) k -  \min\{b, \Pi_{r}(a,\varepsilon) \} ] \ d \mu_0(k,b,\varepsilon)  \\
        & \qquad  - \int \chi_d (1-\chi_{l}) \ \min\{b, (1-\tau) \phi_a (1-\delta) k +\tau \theta V_{cont} (x, \varepsilon)        \} \ d \mu_0(k,b,\varepsilon) \\
        & \qquad  - \int \chi_d (1-\chi_{l}) \  [ \gamma V_{cont} (x, \varepsilon) + \zeta] \ d \mu_0(k,b,\varepsilon) - M \int (1-\chi_{ex}) c_e \ d \Phi(\varepsilon) .
    \end{split}
\end{equation*}

\end{frame}

%------------------------------------------------
\begin{frame}[label=Firms] \frametitle{Heterogeneous Firms - Production \hfill \hyperlink{Model Summary}{\beamergotobutton{Back}}}

Heterogeneous firms: 
\begin{itemize} \small
    \item Capital ($k$), Debt ($b$), Idiosyncratic productivity ($\varepsilon$)
\end{itemize} \normalsize
Firm productivity follows an AR(1) process, with \textit{i.i.d.} innovations \vspace{3mm}

Produce a homogeneous good, using labor $n$ and capital $k$,
$$ y = \varepsilon k^{\alpha}n^{\nu}, \ \ \ \ \alpha,\nu \in (0,1),  \ \nu + \alpha < 1 $$
Every firm $(k,\varepsilon)$ faces the same optimization w.r.t. labor:
$$ \pi(k,\varepsilon) = \max_{n} \ \  \varepsilon k^{\alpha}n^{\nu} - wn - c$$ 
Fixed cost of production: $c$

\end{frame}


%------------------------------------------------
\begin{frame}[label=Firmvalues] \frametitle{Heterogeneous Firms - Values \hfill \hyperlink{Model Summary}{\beamergotobutton{Back}}}

Incumbents may decide to exit, default or continue production: 
\begin{equation*} \label{eq:default decision}
    V_0(k,b,\varepsilon) = \max \{V_{def}, \  V_{exit}(k,b,\varepsilon),  \ V_{cont}(k,b,\varepsilon) \}
\end{equation*}
Default has zero value
$$ V_{def} = 0 $$ 
Voluntarily exit allows to keep the cash on hand, $x$ 
 $$    V_{exit}(k, b, \varepsilon) = \pi(k,\varepsilon) + (1-\delta)k - b  = x $$
Exiting firms are balanced by a mass $M$ of potential entrants 

\end{frame}

%------------------------------------------------
\begin{frame}[label=3] \frametitle{Continuing Firms \hfill \hyperlink{Model Summary}{\beamergotobutton{Back}}}

Continuing firms maximize the discounted sum of dividends $(d)$ choosing
\begin{itemize} \small
    \item Capital $k'$; One-period debt $b'$; CF-based to Total debt, $\tau' \in [0,1]$
\end{itemize} \normalsize


\begin{equation*} \label{eq:V_cont}
    V_{cont}(x,\varepsilon) = \max_{k',b', \tau'} \left( d + q_f \E_{\varepsilon'|\varepsilon} V_0(k',b', \varepsilon') \right)
\end{equation*}
subject to: 
$$ d = x - k' +  \textcolor{red}{q(k',b',\varepsilon, \tau')}b' $$
\begin{equation*} \label{eq:budget}
x' = \pi(k',\varepsilon')+(1-\delta)k'-b' \hspace{5mm} \text{and} \hspace{5mm} d \geq 0
\end{equation*} 

\begin{itemize} \small
    \item where $q(a',\varepsilon) $ is the inverse gross interest rate, and $a' = (k',b',\tau')$
\end{itemize} \normalsize
Continuing firms: no equity finance, $d \geq 0$
 
\end{frame}

%------------------------------------------------
\begin{frame}[label=Entrants] \frametitle{Entrants \hfill \hyperlink{Model Summary}{\beamergotobutton{Back}}}

Every period, there is a mass $M$ of potential entrants
\begin{itemize} \small
    \item Begin with zero cash on hand (asset-poor startups)  
    \item Do not produce, draw initial productivity, $\varepsilon_0 \sim \Phi(\varepsilon)$  
    \item Pay entry cost $c_e$ and continue, or exit within the period, w/o paying $c_e$
\end{itemize} \normalsize \vspace{5mm}

Analogous decision to continuing firms, with zero CoH,
\begin{equation*} \label{eq:entrant decision}
V_e(\varepsilon_0) = \max \{0, \ V_{cont}(0,\varepsilon_0) - c_e \}
\end{equation*}


\end{frame}

Under liquidation, the lender recovers: 
$$   \Pi_{l}(a) =  \min \{b, \ \ (1-\tau) \underbrace{\phi_a (1-\delta) k}_{\textcolor{red}{\text{AB, Liq}}} + \tau \underbrace{\kappa \phi_a  (1-\delta) k}_{\textcolor{red}{\text{CF, Liq}}} \} $$
\begin{itemize} \small
    \item The remainder is retained by the household
\end{itemize} \normalsize
Under reorganization, the lender receives the transfer: 
$$ \Pi_{r}(a,\varepsilon) = \min \{b, \ \  (1-\tau) \underbrace{\phi_a (1-\delta) k}_{\textcolor{red}{\text{AB, Reorg}}} +\tau \underbrace{\theta V_{cont} (x, \varepsilon)}_{\textcolor{red}{\text{CF, Reorg}}} \} $$ 

 
\end{frame}

\begin{frame}[label= interest_rate] \frametitle{The Inverse Interest Rate \hfill \hyperlink{outline2}{\beamergotobutton{Back}}}


The lender considers: \small 
\begin{itemize}\setlength\itemsep{0em} 
    \item Probability of default, $P_D(a', \varepsilon)$
    \item Probability of liquidation in default $\gamma(k',b', \varepsilon)$
    \item Expected payoffs under liquidation and reorganization,  $\Pi_{l}(a')$, $\Pi_{r}(a', \varepsilon)$
\end{itemize} \normalsize

Inverse interest rate ($q$) defined by the zero profit condition:
\begin{equation*} \label{eq:q}
    \begin{split}
        & q(a', \varepsilon)b' = q_0 [ \left(1-P_D(a', \varepsilon)\right)b' +  \\
        & \qquad \qquad P_D(a', \varepsilon) \gamma(a', \varepsilon) \ \Pi_{l}(a') + \\  
        & \qquad  \qquad \underbrace{P_D(a', \varepsilon)}_{\textcolor{red}{\text{Pr(default)}}} (1-\underbrace{\gamma(a', \varepsilon)}_{\textcolor{red}{\text{Pr(Liq.) }}})\ E_{\varepsilon' | \varepsilon} [ \Pi_{r}(a', \varepsilon') |\chi_r = 1 ] \} ]
    \end{split}
\end{equation*}

Continuing firms choose $(k',b',\tau')$, to maximize $V_{cont}(x,\varepsilon)$ s.t. $q(k',b',\tau', \varepsilon)$

\end{frame}

\begin{frame}

\end{frame}

%------------------------------------------------
\begin{frame}[label=LiqReorg_backup] \frametitle{Liquidation and Reorganization \hfill \hyperlink{LiqReorg}{\beamergotobutton{Back}}} 

In default: liquidation or reorganization  \hfill \hyperlink{Default}{\beamergotobutton{Default Probability}} \vspace{3mm}

Liquidation: 
\begin{itemize} \small
    \item Firm exits and capital stock is resold at $\phi_a(1-\delta)k$, where $\phi_a \in (0,1)$  
    \item Lender is recovers $\Pi_l(a)$ any remainder goes to the household
\end{itemize} \normalsize \vspace{3mm} \\

Reorganization: 
\begin{itemize} \small
\item Debt is renegotiated at a variable $\phi_c V_{cont}(x,\varepsilon)$ and a fixed cost $\zeta$ \hfill \hyperlink{FCreorg}{\beamergotobutton{Reorganization Costs}}
\item The lender receives a transfer $\Pi_r(a,\varepsilon)$, financed by equity
\item Firm gets a ‘fresh start’, chooses $(k', b', \tau')$ freely
\end{itemize} \normalsize

Lender’s payoffs $\Pi_l(a)$ and $\Pi_r(a,\varepsilon)$ depend on the debt contract

\end{frame}

%------------------------------------------------
\begin{frame}[label=LiqProb_backup] \frametitle{Liquidation Probability \hfill \hyperlink{LiqReorg}{\beamergotobutton{Back}}} 

Default resolution by bankruptcy courts \vspace{3mm}

Maximizes the value retained after bankruptcy,
\begin{equation*} \label{eq:liquidation decision}
    \max \{ \phi_a (1-\delta) k; \ (1-\phi_c) V_{cont}(x,\varepsilon)- \zeta  \} 
\end{equation*}

Ex-ante, conditional probability of liquidation in default: 
\begin{equation*} \label{eq:liquidation probability}
    \gamma(k', b', \varepsilon) = Pr(\phi_a (1-\delta) k' \ \geq \ (1-\phi_c) V_{cont}(x',\varepsilon') - \zeta | \chi_d = 1)
\end{equation*}  

\textbf{Fixed costs} of reorganization, $\zeta$: decreasing liquidation prob. across firm size

\end{frame}




%------------------------------------------------
\begin{frame}[label = Conversions] \frametitle{Lender's Perspective: Exogenous Risk of Liquidation \hfill  \hyperlink{motivation2}{\beamergotobutton{Back}}}

The debtor is expected to file for bankruptcy
\begin{itemize} \small 
    \item Chapter 7 liquidation or Chapter 11 reorganization
\end{itemize} \normalsize \vspace{6mm} 

Observed Conversions in the US (IDB 2010–2023)
\begin{itemize} \small 
    \item 0.16\% of Chapter 7 cases converted to Chapter 11
\end{itemize} \normalsize  \vspace{6mm}

Legal and practical barriers to reversing liquidation decisions
\begin{itemize} \small
    \item The court is unlikely approve the conversion, if the management is unwilling to continue 
    \item Lender is often ill-equipped to take over the management of the firm
\end{itemize} \normalsize


\end{frame}



%------------------------------------------------
\begin{frame}[label = payoffs_det] \frametitle{Recovery - Further Details \hfill  \hyperlink{payoffs}{\beamergotobutton{Back}}}

\textbf{Near-zero} recovery of CF-based debt under liquidation: 
\begin{itemize} \small
    \item No physical collateral; recovery is costly and time-consuming
    \item Assets are often tied up in other loans (53\% of firms hold both AB and CF-based debt)
    \item Employee wages enjoy priority over unsecured claims - \textit{U.S.C. §507(a)(4)}
\end{itemize} \normalsize \vspace{5mm}

Recovery of AB-based debt under reorganization: 
\begin{itemize} \small
    \item 'Best interest of creditor test': creditors cannot be worse-off under the proposed reorganization plan than they would have been under liquidation - \textit{U.S.C. §1129}
    \item The lender avoids the cost of monitoring uncertain future cash flows
\end{itemize} \normalsize
 \normalsize 


\end{frame}

\begin{frame}[label = FCreorg] \frametitle{Fixed Costs of Reorganization \hfill \hyperlink{motivation2}{\beamergotobutton{Back to Motivation}}  \hyperlink{LiqReorg}{\beamergotobutton{Back To Liq.}}}


Reorganization: lengthy negotiation between debtors and creditors 
\begin{itemize} \small 
    \item Large legal, administrative and time (direct and indirect)  cost
    \item Often, costs of replacing original management
\end{itemize} \normalsize \vspace{3mm} 

\textbf{Fixed cost} of reorganization - \textit{Wang, 2022} 
\begin{itemize} \small 
    \item Smaller firms face higher costs per dollar of assets \\
     \qquad \footnotesize - \textit{LoPucki and Doherty 2004, 2011; Bris et al. 2006} \small 
     \item Smaller firms take longer to reorganize per dollar of assets \footnotesize \qquad - \textit{IDB data} \normalsize 
\end{itemize} \normalsize \vspace{3mm}

Model implementation \qquad  \footnotesize -\textit{Antill and Grenadier 2018; Corbae and D'Erasmo 2021}
\begin{itemize} \small 
    \item Fixed costs replicate declining liquidation risk across firm size \\
\end{itemize} \normalsize

\end{frame}

\begin{frame}[label=Default] \frametitle{Probability of Default \hfill \hyperlink{LiqReorg}{\beamergotobutton{Back}}}
\textbf{Endogenous default} decision:
\begin{equation*} \label{eq:default decision}
    V_0(k,b,\varepsilon) = \max \{V_{def}, \  V_{exit}(k,b,\varepsilon),  \ V_{cont}(k,b,\varepsilon) \}.
\end{equation*}
Let $\chi_{endo}(a', \varepsilon) = 1$ if the firm decides to default \vspace{3mm} \\

\textbf{Exogenous default} shock 
\begin{itemize} \small
    \item With probability $P_{x}$
    \item Matches share of reorganizations to the data
\end{itemize} \normalsize

\textbf{Combined} default probability 
$$ P_D(a', \varepsilon) = E_{\varepsilon'|\varepsilon}[\chi_{endo}(a',\varepsilon')] + P_{x} - E_{\varepsilon'|\varepsilon}[\chi_{endo}(a',\varepsilon')]P_{x} $$


\end{frame}

Small Business Reorganization Act (effective of 2020):
\begin{itemize} \small
    \item Enacted on August 2019, with an effective date of February 2020
    \item Streamline reorganization process up to a debt limit of \$3 million  (\$7.5 million)

\begin{itemize} \small

\end{itemize} \normalsize \vspace{2mm}

\begin{itemize} \small
    \item Simplifies deviations from APR and creditor cramdown
    \item No competing reorganization plan from creditors
\end{itemize} \normalsize
